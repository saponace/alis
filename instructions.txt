// To install a new arch linux


loadkeys fr

create partitions (at least root and swap)
if not boot partition already exist, also create a boot partition of type EFI FILESYSTEM and mkfs.vfat -F 32 this partition
format theses partions (mkfs.ext4 and mkswap)

mount at /mnt and swapon, (do not forget to mount /boot in /mnt/boot)


pacstrap /mnt base base-devel
genfstab -p -L /mnt >> /mnt/etc/fstab


arch-chroot /mnt

vi /etc/locale.gen
    -> uncomment fr_FR... and en_US.UTF-8 lines

pacman -Syu git
git clone https://github.com/saponace/dotfiles.git
init-before-reboot.sh $hostname $username $boot-partition(ex:/dev/sda2) $linux-root-partition(ex:/dev/sda1)


exit

umount and swapoff partitions

reboot

init-all.sh $username
